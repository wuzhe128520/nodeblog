node博客数据库设计
article(文章表)
  类型和长度 是否允许为空 自增 主键 外键     备注
id(文章标识) int(11)   1              
title(文章标题) varchar(50)                  
author(作者) varchar(10)                  
content(内容) longtext                  
time(时间) datetime                  
views(浏览数) int(8)                  
summery(摘要) varchar(300)                  
ding(顶) int(10) 1                
cai(踩) int(10) 1                
img(文章图片地址) varchar(100) 1                
typeid(类型id) int(2)       外键指向articletype          
comments(评论表) (主题字段表示可能是说说、相册、文章的评论)
id(主键) int(11)                  
topic_id(主题id,如文章id,说说id) int(8)             这里存对应主题的主键    
topic_type(主题类型，是文章，还是说说) int(2)             存类型，判断是什么主题下的评论    
content(评论内容) varchar(500)                  
from_uid（写评论的用户id） int(8)                  
reply(回复表)
id(主键) int(11)                  
comment_id(评论ID) int(11)       外键指向comments的主键          
reply_id(被回复的文字的评论id或是回复id)                    
reply_type(回复类型,是回复的回复，还是回复的评论)                    
content(回复内容)                    
from_uid(回复用户id，回复人)                    
to_uid(目标用户id)                    
articletag(文章标签表)
id(主键) int(11)                  
tagname(标签名) varchar(50)                  
aid int(11)       外键(指向article表)          
articletype(文章类型表)
id(主键) int(11)                  
typename(类型名称) varchar(20)                  
user(用户表)
id(主键) int(11)                  
username(用户名) varchar(25)                  
password(密码) varchar(50)                  
email(邮箱) varchar(50)                  
code(激活码) varchar(100)                  
status(激活状态) int(1)                  
admin(是否是管理员) int(1)                  
dic(字典表，中间表)
dicid(主键) int(3)                  
dicvalue(字典的值) varchar(50)                  
groupname(字典含义) varchar(25)                  
say(说说表)
sayid(主键) int(11)                  
userid int(11)                  
content varchar(500)                  
time datetime                  
album(相册表)
albumid(主键) int(11)                  
userid(用户id) int(11)             允许为空    
albumname(相册名称) varchar(50)                  
photos(相片表)
photoid(主键) int(11)                  
albumid(相册id) int(11)                  
time(创建时间) datetime                  
pname(相片名) varchar(50)                  
photosrc(相片地址) varchar(100)                  
thumbanilsrc(缩略图地址) varchar(100)                  
porder(相片顺序) int(3)
